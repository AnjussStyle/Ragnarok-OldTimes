function	script	Func_Thana	{

	if(!countitem(getarg(0)))
	{
		show "[Liei]";
		show "Oh, desculpe-me mas você";
		show "não tem "+ getitemname(getarg(0)) +"";
		show "Por favor check seu inventário novamente...";
		close;
	}
	show "[Liei]";
	show "A recompença para cada";
	show ""+ getitemname(getarg(0)) +" é...";
	show " ";
	show "1,000 zeny";
	show "2,000 EXP";
	next;
	show "[Liei]";
	show "Se você tem "+ countitem(getarg(0)) +"";
	show ""+ getitemname(getarg(0)) +", você";
	show "irá ter um total de...";
	show " ";
	show ""+ countitem(getarg(0)) * 1000 +" zeny";
	show ""+ countitem(getarg(0)) * 2000 +" EXP";
	next;
	show "[Liei]";
	show "Então você gostaria de";
	show "trocar todas as suas "+ getitemname(getarg(0)) +"";
	show "comigo agora?";
	next;
	switch( select( "Sim","Não" ) )
	{
		case 1:
			show "[Liei]";
			show "Boa! Aqui está";
			show ""+ countitem(getarg(0)) * 1000 +" zeny e";
			show ""+ countitem(getarg(0)) * 2000 +" EXP. Obrigado,";
			show "e continue com o";
			show "excelente trabalho~";
			set Zeny,Zeny + countitem(getarg(0)) * 1000;
			getexp countitem(getarg(0))*2000,0;
			delitem getarg(0),countitem(getarg(0));
			close;
			
		case 2:
			show "[Liei]";
			show "Certo, sem problemas.";
			show "Basta voltar a falar comigo";
			show "assim que você desejar";
			show "ter a sua recompença,";
			show "ok?";
			close;
	}
}
