prontera.gat,142,174,5	script	Evento de Natal::Evento de Natal	718,{


	show "[Papail Noel]";
	show "Olá "+ strcharinfo(0) +" !";
	show "Gostaria de fazer o que ?";
	next;
	menu "Manipular Evento",GM_Menu,"Menu dos Players",Player_Menu,"Nada!",-;
	close;

L_Sry:
	show "[Papail Noel]";
	show "Sinto muito "+ strcharinfo(0) +"!";
	show "Somente GMs podem manipular o Evento !";
	close;

Player_Menu:
	show "[Papail Noel]";
	show "Olá "+ strcharinfo(0) +"!";
	show "Gostaria de fazer o que ?";
	next;
	menu "Contar Meus Pontos",L_Count,"Pegar meu Prêmio !",L_Exchange,"Nada...",-;
	close;

L_Count:
	show "[Papail Noel]";
	show ""+ strcharinfo(0) +"";
	show "Você tem "+ @Mob_Killed +" pontos !";
	next;
	goto Player_Menu;

L_Exchange:
	show "[Papail Noel]";
	show "Escolha seu prêmio !";
	next;
	menu "1 Old Blue Box [^00FF0070 Pontos^000000]",L_OBB,
	     "1 Gift Box [^00FF0050 Pontos^000000]",L_GB,
	     "1 Gorro de Natal [^00FF0025 Pontos^000000]",L_GN,
	     "50 Candy Cane [^00FF0020 Pontos^000000]",L_CC,
	     "75 Well-baked Cookie [^00FF0015 Pontos^000000]",L_WBC,
	     "100 Lucky Candy  [^00FF0010 Pontos^000000]",L_LC,
	     "Nenhum",L_Nada;

L_OBB:
	if(@Mob_Killed < 70){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 603,1;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -70;
	close;

L_GB:
	if(@Mob_Killed < 50){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 644,1;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -50;
	close;

L_GN:
	if(@Mob_Killed < 25){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 2236,1;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -25;
	close;

L_CC:
	if(@Mob_Killed < 20){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 530,50;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -20;
	close;

L_WBC:
	if(@Mob_Killed < 15){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 538,75;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -15;
	close;

L_LC:
	if(@Mob_Killed < 10){
	show "[Papail Noel]";
	show "Você não tem pontos suficientes";
	close;
	}
	show "[Papail Noel]";
	show "Muito Bem, vamos pegar o seu prêmio !";
	additem 570,100;
	next;
	show "Aí está !";
	set @Mob_Killed,@Mob_Killed -10;
	close;

L_Nada:
	close;

GM_Menu:
	if (getgmlevel() < 80) goto L_Sry;
	show "[Papail Noel]";
	show "Olá "+ strcharinfo(0) +" !";
	show "Gostaria de fazer o que com o Evento ?";
	next;
	L_Menu:
	menu "Vamos Começar !",L_Begin,"Parar o Evento !",L_Stop;

L_Begin:
	show "[Papail Noel]";
	show "Ok !";
	next;
	announce "O GM "+ strcharinfo(0) +" deu Início ao Evento de Natal!",8;
	next;
	killmonster "job_knt.gat", "KntLvl1::OnMyMobDead";
	atcommand strcharinfo(0)+":@skilloff "+getcharid(0);
	donpcevent "ChristEvent::OnStart";
	initnpctimer "TimerEvent";
	close;

L_Stop:
	announce "O GM "+ strcharinfo(0) +" finalizou o Evento de Natal!",8;
	stopnpctimer;
	killmonsterall "prontera.gat";
	atcommand strcharinfo(0)+":@skillon "+getcharid(0);
	close;
}

// ========================================================
// ======================= Monsters =======================
// ========================================================
prontera.gat,1,1,1	script	ChristEvent	-1,{

OnStart:

	set $@MapMobs, 250;
	monster "prontera.gat",156,175,"Antônio",1247,50,"ChristEvent::OnMobDead";
	monster "prontera.gat",159,172,"Cookie",1246,50,"ChristEvent::OnMobDead";
	monster "prontera.gat",156,168,"Poring Noel",1062,50,"ChristEvent::OnMobDead";
	monster "prontera.gat",152,171,"Presente",1249,50,"ChristEvent::OnMobDead";
	monster "prontera.gat",156,171,"Goblin Natalino",1245,50,"ChristEvent::OnMobDead";
	end;

OnMobDead:
	set @Mob_Killed,@Mob_Killed +1;
	set $@MapMobs,$@MapMobs -1;
	if($@MapMobs == 0) goto Over;
	if($@MapMobs > 0) end;
	stopnpctimer "TimerEvent";
	end;

Over:
	announce "Todos os Monstros foram Mortos, portanto o Evento acabou!",bc_blue|bc_all;
	stopnpctimer "TimerEvent";
	end;
}

prontera.gat,1,1,1	script	TimerEvent	-1,{


OnTimer2000:
	set $@EvtTime$, "Vocês tem 3 minutos para matar todos os monstros";
	callsub EventAnnounce;

OnTimer62000:
	set $@EvtTime$, "Vocês tem 2 minutos para matar todos os monstros";
	callsub EventAnnounce;

OnTimer92000:
	set $@EvtTime$, "Matandos os Monstros você ganha pontos e pode trocar por prêmios";
	callsub EventAnnounce;

OnTimer122000:
	set $@EvtTime$, "Vocês tem 1  minutos para matar todos os monstros";
	callsub EventAnnounce;

OnTimer152000:
	set $@EvtTime$, "Vocês só tem 30 segundos para matar todos os monstros";
	callsub EventAnnounce;

OnTimer162000:
	set $@EvtTime$, "Vocês só tem 10 segundos para matar todos os monstros";
	callsub EventAnnounce;

OnTimer182000:
	set $@EvtTime$, "O Tempo do Evento acabou, vá trocar seus prêmios";
	callsub EventAnnounce;
	killmonsterall "prontera.gat";
	stopnpctimer "TimerEvent";
	end;

EventAnnounce:
	announce "[ Papai Noel ]: "+$@EvtTime$+"!",bc_blue|bc_all;
	end;
}
