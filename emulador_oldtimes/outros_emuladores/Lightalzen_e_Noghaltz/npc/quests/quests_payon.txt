		show "^5555FF1 Runi Amaldiçoado,";
		show "1 Fone de Ouvido,";
		show "200 Plumas,";
		show "e 5,000 zeny ^000000 para que eu possa fazê-lo.";
		next;
		show "[Senhora Desconhecida]";
		show "Com esses itens eu posso fazer o 'Aquecedores de Orelha'.  Acredite em mim, são os melhores e mais bem feitos e ficam bem em qualquer um!";
		close;
	
	M_Muffs:

		show "[Senhora Desconhecida]";
		show "Oh, Pelo que vejo, você tem os itens.  Só um momento...";
		next;
		show "[Senhora Desconhecida]";
		show "... Isso vai aqui... e isso aqui...... cola ali... um pouquinho mais pra cá..........";
		next;
		additem 2283,1;
		delitem 724,1;
		delitem 5001,1;
		delitem 949,200;
		set Zeny, Zeny-5000;
		show "[Senhora Desconhecida]";
		show "Ah, aqui está! Os famosos Aquecedores de Orelha da minha familia.  Aproveite!";
		close;

			show "[Senhora Desconhecida]";
			show "Sintom muito, mas você não tem os itens necessário para que eu possa fabricar os Aquecedores de Orelha da minha familia.";
			close;
	M_End:
		close;

}

		next;
		menu "-Elmo Angelical",sM_Angel, "-Chapéu de Deviruchi ",sM_Devir, "-Eu volto depois.",M_End;

		sM_Angel:
			show "[Jovem]";
			show "Para o Elmo ANgelical você precisa:";

			show "- ^5555FF5 Presa de Hati^000000.";

			show "- ^5555FF40 Garra de Grifo^000000.";
