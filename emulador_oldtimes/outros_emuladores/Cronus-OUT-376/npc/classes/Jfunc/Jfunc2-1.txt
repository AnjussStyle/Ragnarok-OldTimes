//Knight Check

function	script	Kni_check	{

		if(BaseJob != Job_Swordman) goto L_NotSwo;
		show "Bem, primeiro, por favor preencha essa ficha....";
		next;
		show "...(Você preenche a ficha e a entrega de volta)...";
		next;
		show "[Chefe dos Cavaleiros]";
		if(JobLevel < 40) goto L_NotLvl;
		show "Impressionante. Você tem as qualidades para se tornar um Cavaleiro..., porém, precisará muito mais que isso para realmente se tornar um.";
		next;
		return;

		L_NotLvl:
			show "Humm...aparentemente você ainda não alcançou o nivél 40. Poderia fazer a gentileza de retornar quando isso ocorrer.  Eu estarei esperando, bem aqui.";
			close;
		L_NotSwo:
			if(BaseJob == Job_Knight) goto sL_JobKni;
			if(BaseJob == Job_Novice) goto sL_JobNov;

			sL_JobOther:
				show "Me perdoe mas, somente Espadachim podem se tornar Cavaleiros. Tenha um bom dia!";
				close;

			sL_JobKni:
				show "...(sighs)...O que você ainda está fazendo aqui?? Não perca mais tempo!! E continue proteguendo Rune Midgard, meu jovem Cavaleiro.";
				emotion 32;
				close;

			sL_JobNov:
				show "Hahahaha!! Meu jovem, você não tem ainda experiência o suficiente para erguer uma espada quanto mais para se vir aqui perguntar se você pode ser um Cavaleiro. Hahahaha!!";
				emotion 18;

				show "Meu jovem, você primeiro deve ser tornam um Espadachim de fibra para vir se tornar um Cavaleiro. Agora, treine bastante meu pequeno novice. Que um dia você consegue!";
				close;


}
