function	script	F_GldDunSw	{

	show "[ Voz Misteriosa ]";
	show " ' Somente os guerreiros bravos podem conduzir à base do clã .. '";
	next;
	show "(Há um interruptor por aqui.......   você gostaria de puxá-lo? )";
	next;
	menu "Sim",-,"Não",M_No;

		set @GID, GetCastleData(getarg(0)+".gat",1);
		if (@GID == 0 || getcharid(2) != @GID) goto L_NotGLead;
		warp "gld_dun"+getarg(1)+".gat",getarg(2),getarg(3);
		end;

		L_NotGLead:
			show "[ Voz Misteriosa ]";
			show " ' ... parece que você não é bravo bastante... ' ";
	M_No:
		return;
}
