//Acolyte Check

function	script	Aco_check	{
		if(BaseJob == Job_Acolyte) goto JobAco;
		if(BaseJob != Job_Novice) goto NotNov;
		if(JobLevel < 10) goto JobLess10;
		return;

		JobAco:
			show "Você se sente bem?....";
			emotion 20;
			next;
			show "[Bispo Tomas Cerbantes]";
			show "Você realmente já é um "+callfunc("classname",4,"S","M")+".....";
			next;
			show "[Bispo Tomas Cerbantes]";
			show "Intão, não posso fazer nada por você, até mais.";
			emotion 18;
			close;
			

		NotNov:
			show "Sinto muito filho, você já vive para outra profissão.  Sinto muito, mas você não poderá servir a Deus.";
			close;
			

		JobLess10:
			show "Minha criança você não possui ainda as habilidades necessárias para servir ao senhor.";
			show "Volte por favor quando sua fé em Deus se tornar mais forte.";
			next;
			show "[Bispo Tomas Cerbantes]";
			show "Que deus te abençoes.";
			close;
			
}

//Archer Check

function	script	Arc_check	{

		if(Class == 4004) goto sL_Archer;
		if(BaseJob != Job_Novice) goto sL_Other;
		show "Você esta fazendo uma ótima escolha de profissão.";
		show "Me de por favor seu resumo de modo que eu possa verificar suas qualificações.";
		next;
		show "(Você entrega seu resumo)";
		next;
		show "[Guia de "+callfunc("classname",3,"S","M")+"]";
		if(JobLevel < 10) goto sL_LowLvl;
		set job_archer_q,1;
		return;

		sL_Archer:
			show "Seus espirito narulialista me diz que você já é um "+callfunc("classname",3,"S","M")+"....também me diz o arco e flechas em suas mãos...Haha";
			emotion 18;
			close;
			

		sL_Other:
			show "Desculpe, você já tem uma profissão e não pode virar um "+callfunc("classname",3,"S","M")+"...embora eu conpreenda o porque você quer mudar de profissão."; 
			close;
			

		sL_LowLvl:
			show "Parece que você ainda não pode se tornar um "+callfunc("classname",3,"S","M")+".  Por favor olhe os requerimentos mais uma vez.";
			close;
			

}
